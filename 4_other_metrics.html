        'polynomial_3', 'polynomial_5', 'polynomial_7', 'matern', 'rq', 'rbf']

    sub_df = df_2weeks[['time', 'ID', method]]

    # Add the results to the dictionary with method name as key
    results_dict[method] = cgm.all_metrics(sub_df, 'time', 'glc', 'ID',
